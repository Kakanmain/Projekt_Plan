43 void push_back(t val) {
44 Node<t>* newNode = new Node<t>(val);
45
46 if (tail == nullptr) {
47 head = tail = newNode;
48 }
49 else {
50 tail->next = newNode;
51 tail = newNode;
52 }
53 }
54
56 void remove(t val) {
57 if (head == nullptr) return;
58
59 if (head->value == val) {
60 Node<t>* temp = head;
61 head = head->next;
62 if (temp == tail) {
63 tail = nullptr;
64 }
65 delete temp;
66 return;
67 }
68
69 Node<t>* current = head;
70 while (current->next != nullptr && current->next->value != val) {
71 current = current->next;
72 }
73
74 if (current->next != nullptr) {
75 Node<t>* temp = current->next;
76 current->next = temp->next;
77 if (temp == tail) {
78 tail = current;
79 }
80 delete temp;
81 }
82 }
83
84 ~lista() {
85 Node<t>* current = head;
86 while (current != nullptr) {
87 Node<t>* next = current->next;
88 delete current;
89 current = next;
90 }
91 }
